def taiju(data)
  sum_of_men=0
  sum_of_women=0
  num_of_men=0
  num_of_women=0
  data.each do |d|
    if d[0]=="男" then
      sum_of_men=sum_of_men+d[1]
      num_of_men=num_of_men+1
    else
      sum_of_women=sum_of_women+d[1]
      num_of_women=num_of_women+1
    end
  end
  return [sum_of_men.to_f/num_of_men, sum_of_men.to_f/num_of_men]
end

p taiju(data)
